 63        intersects: List[np.ndarray] = []

 65            for plane_normal in unit_vecs.T:
 66                p = intersect_line_plane(
 67                    out_position[:3], out_normal[:3], plane_origin[:3], plane_normal[:3]
 68                )
 69                if p is not None:
 70                    intersects.append(p)
 71        return as_sampler_points(intersects)
 72
 73    intersects = _find_intersections()
 74    intersects_trans = np.dot(affine_inv, intersects)
 75    intersects_in_bounds = np.all(
 76        ((vecs_origin - 1e-6) < intersects_trans)
 77        & (intersects_trans < (vecs_datalim + 1e-6)),
 78        axis=0,
 79    )
